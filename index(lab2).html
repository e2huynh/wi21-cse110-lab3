void HCTree::build(vector<unsigned char>& rep, unordered_set<int> symbolIndex) {
    queue<int> q; //q and nodes are parallel queues
    queue<HCNode*> nodes;
    root = new HCNode();
    root -> symbol = rep[1];
    q.push(1);
    nodes.push(root);

    //construct the tree in a level-order traversal
    while(!q.empty()) {
        int index = q.front();
        HCNode* temp = nodes.front();
        q.pop();
        nodes.pop();

        //if an actual symbol
        if(symbolIndex.count(index))
            leaves[rep[index]] = temp;

        if(index * 2 < rep.size()) {
            temp -> c0 = new HCNode();
            temp -> c0 -> symbol = rep[index*2];
            temp -> c0 -> p = temp;

            q.push(index*2);
            nodes.push(temp -> c0);
        }

        if(index*2 + 1 < rep.size()) {
            temp -> c1 = new HCNode();
            temp -> c1 -> symbol = rep[index*2 + 1];
            temp -> c1 -> p = temp;

            q.push(index*2 + 1);
            nodes.push(temp -> c1);
        }
    }
    
    //load the huffman code for each leaf
    for(HCNode* l : leaves) {
        if(l) {
            string binaryCode = "";
            HCNode* curr = l -> p;
            HCNode* prev = l;

            while(curr) {
                //if prev is the right child of curr
                if(curr -> c1 == prev)
                    binaryCode = "1" + binaryCode;
                else
                    binaryCode = "0" + binaryCode;

                prev = curr;
                curr = curr -> p;
            }

            binaryMap[binaryCode] = l -> symbol; //use map for decoding because we search for encodings a lot
        }
    }
}
